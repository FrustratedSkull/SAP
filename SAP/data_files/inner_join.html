TYPES: BEGIN OF TY_VBAK,
          VBELN TYPE VBAK-VBELN,
          ERDAT TYPE VBAK-ERDAT,
          ERNAM TYPE VBAK-ERNAM,
          POSNR TYPE VBAP-POSNR,
          MATNR TYPE MARA-MATNR,
          MTART TYPE MARA-MTART,
          MBRSH TYPE MARA-MBRSH,
        END OF TY_VBAK.


  DATA: GT_VBAK TYPE STANDARD TABLE OF TY_VBAK,
        GS_VBAK TYPE TY_VBAK.

SELECT-OPTIONS SO_VBELN FOR GS_VBAK-VBELN.

SELECT  VBAK~VBELN VBAK~ERDAT VBAK~ERNAM
        VBAP~POSNR VBAP~MATNR
        MARA~MTART MARA~MBRSH
        INTO TABLE GT_VBAK
        FROM VBAK
        INNER JOIN VBAP ON VBAK~VBELN = VBAP~VBELN
        INNER JOIN MARA ON VBAP~MATNR = MARA~MATNR
        WHERE VBAK~VBELN IN SO_VBELN.


LOOP AT GT_VBAK INTO GS_VBAK.

          WRITE:/ GS_VBAK-VBELN, GS_VBAK-ERDAT, GS_VBAK-ERNAM,
                  GS_VBAK-MATNR, GS_VBAK-POSNR,
                  GS_VBAK-MBRSH, GS_VBAK-MTART.

ENDLOOP.
